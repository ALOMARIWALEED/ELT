export async function upsertUser(user: InsertUser): Promise {
  if (!user.openId) {
    throw new Error("User openId is required for upsert");
  }

  const db = await getDb();
  if (!db) {
    console.warn("[Database] Cannot upsert user: database not available");
    return;
  }

  try {
    const values: InsertUser = {
      openId: user.openId,
    };
    const updateSet: Record = {};

    const textFields = ["name", "email", "loginMethod"] as const;
    type TextField = (typeof textFields)[number];

    const assignNullable = (field: TextField) => {
      const value = user[field];
      if (value === undefined) return;
      const normalized = value ?? null;
      values[field] = normalized;
      updateSet[field] = normalized;
    };

    textFields.forEach(assignNullable);

    if (user.lastSignedIn !== undefined) {
      values.lastSignedIn = user.lastSignedIn;
      updateSet.lastSignedIn = user.lastSignedIn;
    }
    if (user.role !== undefined) {
      values.role = user.role;
      updateSet.role = user.role;
    } else if (user.openId === ENV.ownerOpenId) {
      values.role = "admin";
      updateSet.role = "admin";
    }

    if (!values.lastSignedIn) {
      values.lastSignedIn = new Date();
    }

    if (Object.keys(updateSet).length === 0) {
      updateSet.lastSignedIn = new Date();
    }

    await db.insert(users).values(values).onDuplicateKeyUpdate({
      set: updateSet,
    });
  } catch (error) {
    console.error("[Database] Failed to upsert user:", error);
    throw error;
  }
}

export async function updateKPIMetrics(classId: number) {
  const db = await getDb();
  if (!db) return;

  const classStudents = await db
    .select()
    .from(students)
    .where(eq(students.classId, classId));

  if (classStudents.length === 0) return;

  const totalStudents = classStudents.length;
  const averageScore =
    classStudents.reduce((sum, s) => sum + Number(s.currentScore || 0), 0) /
    totalStudents;

  const passedCount = classStudents.filter(
    (s) => Number(s.currentScore || 0) >= 65
  ).length;
  const successRate = (passedCount / totalStudents) * 100;

  const atRiskCount = classStudents.filter((s) => s.isAtRisk).length;

  // Calculate Mission Readiness Index
  const completionRate =
    classStudents.filter((s) => (s.totalExamsTaken || 0) > 0).length / totalStudents;
  const missionReadinessIndex =
    (successRate * 0.4 + averageScore * 0.3 + completionRate * 100 * 0.3) / 100;

  // Find weakest and strongest skills
  const skillPerformances = await db.select().from(skillPerformance);
  const avgBySkill: Record = {};

  for (const sp of skillPerformances) {
    if (!avgBySkill[sp.skill]) {
      avgBySkill[sp.skill] = 0;
    }
    avgBySkill[sp.skill] += Number(sp.averageScore || 0);
  }

  const weakestSkill = Object.entries(avgBySkill).sort(
    (a, b) => a[1] - b[1]
  )[0]?.[0];
  const strongestSkill = Object.entries(avgBySkill).sort(
    (a, b) => b[1] - a[1]
  )[0]?.[0];

  const existing = await db
    .select()
    .from(kpiMetrics)
    .where(eq(kpiMetrics.classId, classId))
    .limit(1);

  if (existing.length > 0) {
    await db
      .update(kpiMetrics)
      .set({
        totalStudents,
        averageScore: averageScore.toString(),
        successRate: successRate.toString(),
        completionRate: (completionRate * 100).toString(),
        weakestSkill,
        strongestSkill,
        atRiskStudents: atRiskCount,
        missionReadinessIndex: missionReadinessIndex.toString(),
        lastUpdated: new Date(),
      })
      .where(eq(kpiMetrics.id, existing[0].id));
  } else {
    const kpiValues: any = {
      classId,
      totalStudents,
      averageScore: averageScore.toString(),
      successRate: successRate.toString(),
      completionRate: (completionRate * 100).toString(),
      weakestSkill,
      strongestSkill,
      atRiskStudents: atRiskCount,
      missionReadinessIndex: missionReadinessIndex.toString(),
    };
    await db.insert(kpiMetrics).values(kpiValues);
  }
}
